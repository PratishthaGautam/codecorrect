Before Refactoring

function sum(a, b, c, d, e) {
    var s = 0;
    s = a + b;
    s = s + c;
    s = s + d;
    s = s + e;
    return s;
}

After Refactoring

function calculateSum(numbers) {
    return numbers.reduce((total, num) => total + num, 0);
}

  • Replaced multiple additions with reduce() for cleaner logic.
